$ if "''P1'" .eqs. "LINK" then goto linkup
$ call compile lr
$ call compile lrcalc
$ call compile lrcode
$ call compile lrdump
$ call compile lreval
$ call compile lrfree
$ call compile lrglib
$ call compile lrlibr
$ call compile lrload
$ call compile lroptn
$ call compile lrsort
$ call compile lrsymb
$
$ linkup:
$ write sys$output "Linking lr..."
$ link lr,-
       lrcalc,lrcode,lrdump,lreval,lrfree,lrglib,-
       lrlibr,lrload,lroptn,lrsort,lrsymb
$ exit
$
$ compile: subroutine
$ write sys$output "Compiling ''P1'..."
$ cc 'P1'
$ endsubroutine
